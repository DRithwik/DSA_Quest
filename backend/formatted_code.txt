/**
 * Handles the contact form submission by constructing a mailto link.
 * Prevents the default form submission and redirects the user to their email client.
 * @param {Event} event - The submit event from the form.
 */
function sendMail(event) {
  event.preventDefault(); // Prevent form from submitting normally

  // Get form values
  var name = document.getElementById('name').value;
  var email = document.getElementById('email').value;
  var message = document.getElementById('message').value;

  // Construct the mailto URL with subject and body pre-filled.
  // NOTE: This relies on the user's local email client and is not a server-side submission.
  // IMPORTANT: Replace 'your-actual-email@example.com' with the desired recipient email address.
  var recipientEmail = 'your-actual-email@example.com'; 
  var subject = encodeURIComponent('New message from ' + name);
  var body = encodeURIComponent('You have a new message from ' + name + ' (' + email + ')\n\nMessage:\n' + message);

  var mailtoLink = 'mailto:' + recipientEmail + '?subject=' + subject + '&body=' + body;

  // Inform the user before redirecting
  alert('Redirecting to your email client. Please ensure it is configured to send the message.');

  // Redirect to the mailto link
  window.location.href = mailtoLink;
}
</script>
